<template>
  <div>
    <h3>{{curPeople.name}} Details</h3>
    <p>id:{{getData.id}}</p>
    <p>
      <span>name:</span>
    </p>
    <input type="text" class="input" v-model="curPeople.name" />
    <button @click="goback">goback</button>
    <button @click="save">save</button>
    <Message />
  </div>
</template>
 
<script lang="ts">
import { Component, Vue } from "vue-property-decorator";
import store from "@/store/index";
import Message from "@/components/message.vue";
 
@Component({
  components: {
    Message
  }
})
export default class details extends Vue {
  curPeople: object = {};
 
  private get getData(): any {
    let arr = JSON.parse(JSON.stringify(store.getters.getData));
    this.curPeople = arr.filter((x: any) => x.id == this.$route.params.id)[0];
    return this.curPeople;
  }
 
  goback() {
    this.$router.go(-1);
  }
 
  save() {
    store.commit("EDICT_HERO", this.curPeople);
    this.$router.push("/heroes");
  }
 
  mounted() {
    store.commit("ADD_MESSAGE", "added hero id=" + this.$route.params.id);
  }
}
</script>
